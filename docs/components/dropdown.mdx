# Dropdown

### When To Use

When faced with multiple choices, you have the option to enclose them within a Dropdown. By either hovering over or clicking on the trigger, a menu will appear, presenting a list of options for you to select from and carry out the corresponding action.

### Basic

export const SearchInDropdown = ({ children, color }) => {
  const [search, setSearch] = useState();
  const mouseOverHandler = () => {};
  const clickHandler = (e) => {
    e.preventDefault();
  };
  const options = [
    {
      label: "New",
      children: [
        { label: "Folder" },
        {
          label: "File",
          children: [{ label: "From disk" }, { label: "From Onedrine" }],
        },
      ],
    },
    {
      label: "Open",
      children: [
        {
          label: "Recent",
          children: [
            ...[1, 2, 3, 4, 5, 6, 7].map((i) => {
              return {
                label: "App " + i,
              };
            }),
          ],
        },
      ],
    },
    { label: "Close project" },
  ];
  return (
    <>
      <HyDropdown
        search={search}
        placeholder="Select an option"
        options={options}
      >
        {search}
        <HyInput
          placeholder="Type to search"
          slot="label"
          icon="ellipsis-v"
          valueChange={function (e) {
            setSearch(e.detail.value);
          }}
        >
          <span slot="suffix">
            <hy-icon
              name="times-circle"
              click={function () {
                setSearch("");
              }}
            ></hy-icon>{" "}
          </span>
        </HyInput>
      </HyDropdown>
    </>
  );
};

### Custom template

You have the flexibility to utilize a custom template for each option or specify a template for individual options.
The template employs the lit-html package, making it compatible with any framework or library for its usage.

export const BasicDropdowCustomTemplate = ({ children, color }) => {
  const mouseOverHandler = () => {};
  const clickHandler = (e) => {
    e.preventDefault();
    alert("clicked");
  };
  const options = [
    {
      label: "Option 1",
      template: function (item) {
        return html`<div style="display: flex"><hy-input style="width: 80%; text: red !important" value="${item.label} "> </hy-input><hy-button style="width: 19%" @click=${clickHandler} > ok</hy-button></span> </div>`;
      },
    },
    {
      label: "Option 2",
    },
    {
      label: "Option 3",
    },
  ];
  return (
    <>
      <HyDropdown
        placeholder="Select an option"
        options={options}
        trigger="hover"
      >
        <HyButton slot="label" type="link" icon="angle-down">
          Hover me
        </HyButton>
      </HyDropdown>
    </>
  );
};
