# Input

The text field is a basic interface element that is commonly used to receive user input.
Input can be provided or modified using either a keyboard or a mouse.

export const ColorPicket = ({ children, color, cssVariableName }) => {
  const [updateColor, setUpdateColor] = useState(color);
  const [displayPicker, setdisplayPicker] = useState(false);
  const handleChangeComplete = (_color) => {
    setUpdateColor(_color.hex);
    document.documentElement.style.setProperty(cssVariableName, _color.hex);
  };
  const toogleDisplayPicker = () => {
    setdisplayPicker(!displayPicker);
  };
  return (
    <>
      <div
        style={{
          height: "35px",
          width: "35px",
          backgroundColor: updateColor,
          cursor: "pointer ",
          margin: "0 auto 20px",
        }}
        onClick={toogleDisplayPicker}
      ></div>
      {displayPicker && (
        <BlockPicker
          color={updateColor}
          onChangeComplete={handleChangeComplete}
        />
      )}
    </>
  );
};

export const InputPicket = ({ children, value, cssVariableName, unit }) => {
  const [updateValue, setUpdateValue] = useState(value);
  const [displayPicker, setdisplayPicker] = useState(false);
  const handleChangeComplete = (e) => {
    setUpdateValue(e.target.value);
    document.documentElement.style.setProperty(
      cssVariableName,
      `${e.target.value}${unit}`
    );
  };
  const toogleDisplayPicker = () => {
    setdisplayPicker(!displayPicker);
  };
  return (
    <>
      <input
        type="number"
        style={{ width: "60px", display: "inline-block" }}
        onChange={handleChangeComplete}
        value={updateValue}
      />
      <div>{unit}</div>
    </>
  );
};
